    workaround for an asynchronous foreach

    Name Type Description
    array Array array to be iterated
    callback function asynchronous function that is called for each array value
    Returns:
    Promise.<void>
    Example

await asyncForEach( waiting_times, async ( value, i, array ) => {

  console.log( value, i, array );
} );
