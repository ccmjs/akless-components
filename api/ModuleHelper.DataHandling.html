    static exports.arrToObj(obj, keyopt) → {Object.<string, boolean>}
    helper.mjs, line 239

    converts an array of strings or numbers to an object of booleans
    Parameters:
    Name Type Attributes Description
    obj Array | Object array or object that contains the array
    key string <optional>
    		 object property where the array is contained
    Returns:
    Object.<string, boolean> - resulting object
    Examples
    console.log( arrToObj( [ 'foo', 'bar' ] ) );  // => { foo: true, bar: true }
    console.log( arrToObj( [ 1, 2 ] ) );  // => { 1: true, 2: true }
    console.log( arrToObj( { key: [ 'foo', 'bar' ] }, 'key' ) );  // => { foo: true, bar: true }
    const obj = { key: [ 'foo', 'bar' ] };
arrToObj( obj, 'key' );  // without return value (original object is manipulated)
console.log( obj );      // => { key: { foo: true, bar: true } }

    helper.mjs, line 293

    decodes with encodeJSON() encoded JSON

    All %'% are replaced with ".
    Returns passed parameter if decoding fails.
    Parameters:
    Name Type Attributes Default Description
    str string encoded JSON
    ccm Object <optional>
    		 window.ccm used ccm framework version
    Returns:
    Object | Array - decoded JSON
    Examples
    decodeJSON( "{%'%log%'%:true,%'%restart%'%:true}" )       // => { log: true, restart: true }
    decodeJSON( "[%'%ccm.instance%'%,%'%./ccm.user.js%'%]" )  // => [ "ccm.instance", "./ccm.user.js" ]

    static exports.filterData(data, maskopt, ccmopt) → {Object}
    helper.mjs, line 345

    filters a specific subset of complex data
    Parameters:
    Name Type Attributes Default Description
    data Object complex data
    mask Object <optional>
    		 defines specific subset
    ccm Object <optional>
    		 window.ccm used ccm framework version
    Returns:
    Object - subset of complex data
    Examples
    filterData( { a: 'x', b: 'y', c: 'c' }, { a: true, b: true } )  // => { a: 'x', b: 'y' }
    filterData( { a: [ 1, 2, 3 ] }, { 'a.1': true, 'a.2': true } )  // => { a: [ null, 2, 3 ] }
    filterData( { a: [ 1, 2, 3 ] }, { a: true, 'a.2': false } )  // => { a: [ 1, 2, '' ] }
    filterData( { a: { x: 1, y: 2, z: 3 } }, { 'a.x': true, 'a.y': true } )  // => { a: { x: 1, y: 2 } }
