    static exports.privatize(object, propertiesopt) → {Object}
    helper-4.0.0.mjs, line 1077

    privatizes public members of an object

    Deletes all given properties in an object and returns another object with the deleted properties and there values.
    If no properties are given, then all not ccm relevant instance members will be privatized.
    List of ccm relevant properties that will not be privatized:
    • ccm
    • component
    • config
    • element
    • id
    • index
    • parent
    • root
    • shadow
    In addition to this: All functions and depending ccm context relevant ccm instances will also not be privatized. If the first passed property is boolean 'true', than the privatized properties will not be deleted in the passed object.
    Parameters:
    Name Type Attributes Description
    object Object object or ccm instance
    properties string | boolean <optional>
    		 properties that have to privatized
    Returns:
    Object - object that contains the privatized properties and there values
    Examples
    // privatize two public instance members
ccm.component( {
  name: 'dummy1',
  ccm: ccm,
  config: { foo: 'abc', bar: 'xyz', baz: 4711 },
  Instance: function () {
    let my;
    this.ready = async () => {
      my = ccm.helper.privatize( this, 'foo', 'bar' );
      console.log( my );                // => { foo: 'abc', bar: 'xyz' }
      console.log( my.foo, this.foo );  // => abc undefined
      console.log( my.bar, this.bar );  // => xyz undefined
      console.log( my.baz, this.baz );  // => undefined 4711
    };
  }
} );
    // privatize all possible public instance members
ccm.component( {
  name: 'dummy2',
  ccm: ccm,
  config: { foo: 'abc', bar: 'xyz', baz: 4711 },
  Instance: function () {
    let my;
    this.ready = async () => {
      my = ccm.helper.privatize( this );
      console.log( my );                // => { foo: 'abc', bar: 'xyz', baz: 4711 }
      console.log( my.foo, this.foo );  // => abc undefined
      console.log( my.bar, this.bar );  // => xyz undefined
      console.log( my.baz, this.baz );  // => 4711 undefined
    };
  }
} );
    // log all non-ccm specific instance members
ccm.component( {
  name: 'dummy3',
  ccm: ccm,
  config: { foo: 'abc', bar: 'xyz', baz: 4711, logger: [ 'ccm.instance', logger_component_url, logger_config ] },
  Instance: function () {
    this.ready = async () => {
      this.logger && this.logger.log( 'ready', $.privatize( this, true ) );
      console.log( this.foo, this.bar, this.foo );  // => abc xyz 4711
    };
  }
} );

    static exports.protect(html) → {string|Element}
    helper-4.0.0.mjs, line 1121

    filters script elements out of given HTML
    Parameters:
    Name Type Description
    html string | Element HTML String or HTML Element
    Returns:
    string | Element - cleaned HTML
    Examples
    protect( "Hello <script>alert('XSS');</script>World!" ) // => 'Hello World!'
    // <div>Hello <script>alert('XSS');</script>World!</div>
div = protect( div ); // => <div>Hello, World!</div>
